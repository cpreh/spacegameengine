set(
	SGE_INSTALL_EXAMPLE_DIR
	"${CMAKE_INSTALL_BINDIR}"
	CACHE
	STRING
	"Custom example installation directory"
)

function(
	add_sge_example_base
	PATH_NAME
)
	set(
		SINGLE_ARGS
		CONSOLE_EXAMPLE
	)

	set(
		MULTI_ARGS
		SGE_DEPS
		ADDITIONAL_DEPS
		COMPILE_OPTIONS
	)

	cmake_parse_arguments(
		""
		""
		"${SINGLE_ARGS}"
		"${MULTI_ARGS}"
		${ARGN}
	)

	sge_check_optional_deps(
		"The ${PATH_NAME} example"
		"${_SGE_DEPS}"
	)

	string(
		REPLACE
		"/"
		"_"
		FULL_NAME
		${PATH_NAME}
	)

	set(
		FULL_NAME
		"sge_example_${FULL_NAME}"
	)

	set(
		SOURCE_FILE
		"${FCPPT_UTILS_PROJECT_SOURCE_DIR}/examples/${PATH_NAME}.cpp"
	)

	if(
		_CONSOLE_EXAMPLE
	)
		add_executable(
			${FULL_NAME}
			${SOURCE_FILE}
		)
	else()
		awl_utils_add_portable_executable(
			${FULL_NAME}
			"example_main"
			${SOURCE_FILE}
		)

		target_link_libraries(
			${FULL_NAME}
			PRIVATE
			${awl_TARGET}
		)
	endif()

	fcppt_utils_set_target_folder(
		${FULL_NAME}
		examples
		${PATH_NAME}
	)

	fcppt_utils_set_target_compiler_flags(
		${FULL_NAME}
	)

	transform_sge_link_targets(
		"${_SGE_DEPS}"
		SGE_LIBRARIES_RESULT
	)

	target_compile_options(
		"${FULL_NAME}"
		PRIVATE
		${_COMPILE_OPTIONS}
	)

	target_include_directories(
		"${FULL_NAME}"
		PRIVATE
		${FCPPT_UTILS_PROJECT_SOURCE_DIR}/examples/include
	)

	target_link_libraries(
		${FULL_NAME}
		PRIVATE
		${_ADDITIONAL_DEPS}
		${SGE_LIBRARIES_RESULT}
	)

	install(
		TARGETS
		${FULL_NAME}
		DESTINATION
		"${SGE_INSTALL_EXAMPLE_DIR}"
	)
endfunction()
