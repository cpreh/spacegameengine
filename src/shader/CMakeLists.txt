MACRO(
	ADD_SGE_SHADER_INCLUDE_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"include\\\\\\\\sge\\\\\\\\shader" #cmake bitch
		${CMAKE_SOURCE_DIR}/include/sge/shader
		"${SOURCEPATH}"
		*.hpp
		TEMPFILES
	)

	SET(SGESHADERFILES ${SGESHADERFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_SHADER_INCLUDE_SOURCE_GROUP("")

MACRO(
	ADD_SGE_SHADER_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"src\\\\\\\\shader"
		""
		"${SOURCEPATH}"
		*.?pp
		TEMPFILES	
	)

	SET(SGESHADERFILES ${SGESHADERFILES} ${TEMPFILES})
ENDMACRO()

ADD_SGE_SHADER_SOURCE_GROUP("")
