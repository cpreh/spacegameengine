MACRO(
	ADD_SGE_SPRITE_INCLUDE_SOURCE_GROUP
	SOURCEPATH
)
	ADD_SGE_SOURCE_GROUP(
		"include\\\\\\\\sge\\\\\\\\sprite" #cmake bitch
		${CMAKE_SOURCE_DIR}/include/sge/sprite
		"${SOURCEPATH}"
		*.hpp
		TEMPFILES
	)

	SET(SGESPRITEFILES ${SGESPRITEFILES} ${TEMPFILES})
ENDMACRO()
