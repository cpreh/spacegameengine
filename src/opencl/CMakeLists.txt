	OpenGL
)

	SYSTEM
	${OPENGL_INCLUDE_DIR}
)

	WIN32
)
	set(
		SGE_OPENCL_HAVE_WINDOWS TRUE
	)

	find_file(
		GLX_HEADER
		glx.h
		PATHS "${OPENGL_INCLUDE_DIR}/GL"
	)

	if(
		GLX_HEADER
	)
		set(
			SGE_OPENCL_HAVE_GLX TRUE
		)
	endif()
endif()
