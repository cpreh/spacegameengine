MACRO(
	ADD_SGE_BASE_LIBRARY
	LIB_NAME
	ADDITONAL_DEPS
)
	SET(
		SGE_LIB_NAME
		sge${LIB_NAME}
	)

	STRING(
		REPLACE
		"_"
		""
		SGE_LIB_NAME
		${SGE_LIB_NAME}
	)

	STRING(
		TOUPPER
		${LIB_NAME}
		UPPER_LIB_NAME
	)

	APPEND_SOURCE_DIR_AND_MAKE_GROUPS(
		"${SGE_${UPPER_LIB_NAME}_FILES}"
		SGE_${UPPER_LIB_NAME}_FILES_ABS
	)

	ADD_LIBRARY(
		${SGE_LIB_NAME}
		SHARED
		${SGE_${UPPER_LIB_NAME}_FILES_ABS}
	)

	TARGET_LINK_LIBRARIES(
		${SGE_LIB_NAME}
		sgecore
		${ADDITONAL_DEPS}
	)

	INSTALL(
		TARGETS
		${SGE_LIB_NAME}
		DESTINATION
		lib
	)

	STRING(
		REPLACE
		"_"
		"/"
		LIB_INCLUDE_PATH
		${LIB_NAME}
	)

	GET_FILENAME_COMPONENT(
		LIB_INCLUDE_PATH_BASE
		${LIB_INCLUDE_PATH}
		PATH
	)

	INSTALL(
		DIRECTORY
		${CMAKE_SOURCE_DIR}/include/sge/${LIB_INCLUDE_PATH}
		DESTINATION
		include/sge/${LIB_INCLUDE_PATH_BASE}
	)
ENDMACRO()
